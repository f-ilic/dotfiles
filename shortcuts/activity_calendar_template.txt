  <script>
    (function () {
      "use strict";

      var palette = [
        { stroke: "#2d6cdf", fill: "#dce9ff" },
        { stroke: "#c23b73", fill: "#f9ddea" },
        { stroke: "#21805f", fill: "#d9f0e8" },
        { stroke: "#d47113", fill: "#fde5c7" },
        { stroke: "#7657c6", fill: "#e9e2fb" },
        { stroke: "#0f7d8c", fill: "#d7eef2" },
        { stroke: "#b24b35", fill: "#f4ddd8" },
        { stroke: "#527a21", fill: "#e3efce" }
      ];

      var rawCsv = document.getElementById("csv-data").textContent.trim();
      var today = new Date();
      var currentYear = today.getFullYear();
      var todayKey = keyFromDate(today);
      var entries = parseCsv(rawCsv)
        .map(function (cells) {
          return {
            date: String(cells[0] || "").trim(),
            value: cells.slice(1).join(",").trim()
          };
        })
        .filter(function (entry) {
          return /^\d{4}-\d{2}-\d{2}$/.test(entry.date) && entry.value;
        })
        .sort(function (a, b) {
          return a.date.localeCompare(b.date);
        });

      var values = [];
      var valueSeen = Object.create(null);
      entries.forEach(function (entry) {
        if (!valueSeen[entry.value]) {
          valueSeen[entry.value] = true;
          values.push(entry.value);
        }
      });

      var colorByValue = Object.create(null);
      values.forEach(function (value, index) {
        colorByValue[value] = palette[index % palette.length];
      });

      var selectedYear = currentYear;

      document.getElementById("prev-year").addEventListener("click", function () {
        selectedYear -= 1;
        renderYear(selectedYear);
      });

      document.getElementById("next-year").addEventListener("click", function () {
        if (selectedYear < currentYear) {
          selectedYear += 1;
          renderYear(selectedYear);
        }
      });

      renderYear(selectedYear);

      function parseCsv(text) {
        if (!text) {
          return [];
        }
        return text.split(/\r?\n/).map(parseCsvLine).filter(function (cells) {
          return cells.some(function (cell) {
            return String(cell).trim();
          });
        });
      }

      function parseCsvLine(line) {
        var cells = [];
        var cell = "";
        var inQuotes = false;

        for (var i = 0; i < line.length; i += 1) {
          var ch = line[i];
          var next = line[i + 1];

          if (ch === "\"" && inQuotes && next === "\"") {
            cell += "\"";
            i += 1;
          } else if (ch === "\"") {
            inQuotes = !inQuotes;
          } else if (ch === "," && !inQuotes) {
            cells.push(cell);
            cell = "";
          } else {
            cell += ch;
          }
        }

        cells.push(cell);
        return cells;
      }

      function renderYear(year) {
        var yearEntries = entries.filter(function (entry) {
          return entryIsVisibleInYear(entry, year);
        });
        var yearValues = valuesForEntries(yearEntries);
        var byDate = entriesByDateFor(yearEntries);
        var endMonthIndex = year === currentYear ? today.getMonth() : 11;

        document.body.classList.toggle("is-empty", yearEntries.length === 0);
        document.querySelector(".empty-message").textContent = "No entries recorded for " + String(year) + ".";

        renderSummary(yearEntries, yearValues, byDate, year);
        renderLegend(yearValues, colorByValue);
        renderMonths(year, endMonthIndex, byDate, colorByValue);
        updateControls(year);
      }

      function renderSummary(yearEntries, yearValues, byDate, year) {
        var dates = Object.keys(byDate);
        var range = year === currentYear ? String(year) + " year to date through " + todayKey : String(year) + " full year";

        document.getElementById("range-summary").textContent = range;

        var stats = document.getElementById("stats");
        stats.textContent = "";
        stats.appendChild(stat(String(yearEntries.length), "entries"));
        stats.appendChild(stat(String(dates.length), "dates"));
        stats.appendChild(stat(String(yearValues.length), "labels"));
      }

      function stat(number, label) {
        var box = el("div", "stat");
        var strong = el("strong", "", number);
        var span = el("span", "", label);
        box.appendChild(strong);
        box.appendChild(span);
        return box;
      }

      function renderLegend(yearValues, colorMap) {
        var legend = document.getElementById("legend");
        legend.textContent = "";
        yearValues.forEach(function (value) {
          var color = colorMap[value];
          var item = el("div", "legend-item");
          var swatch = el("span", "swatch");
          var label = el("span", "legend-label", value);
          swatch.style.backgroundColor = color.fill;
          swatch.style.border = "1px solid " + color.stroke;
          item.title = value;
          item.appendChild(swatch);
          item.appendChild(label);
          legend.appendChild(item);
        });
      }

      function renderMonths(year, endMonthIndex, byDate, colorMap) {
        var months = document.getElementById("months");
        var cursor = new Date(year, 0, 1);
        var end = new Date(year, endMonthIndex, 1);

        months.textContent = "";
        while (cursor <= end) {
          months.appendChild(renderMonth(cursor.getFullYear(), cursor.getMonth(), byDate, colorMap));
          cursor = new Date(cursor.getFullYear(), cursor.getMonth() + 1, 1);
        }
      }

      function renderMonth(year, monthIndex, byDate, colorMap) {
        var month = el("article", "month");
        var head = el("div", "month-head");
        var title = el("h2", "month-title", monthName(year, monthIndex));
        var count = monthEntryCount(year, monthIndex, byDate);
        var countLabel = el("span", "month-count", count ? String(count) : "");
        var weekdays = el("div", "weekdays");
        var grid = el("div", "grid");
        var firstDay = new Date(year, monthIndex, 1);
        var offset = (firstDay.getDay() + 6) % 7;
        var daysInMonth = new Date(year, monthIndex + 1, 0).getDate();

        ["M", "T", "W", "T", "F", "S", "S"].forEach(function (name) {
          weekdays.appendChild(el("div", "weekday", name));
        });

        for (var i = 0; i < offset; i += 1) {
          grid.appendChild(el("div", "day empty"));
        }

        for (var day = 1; day <= daysInMonth; day += 1) {
          var key = dateKey(year, monthIndex + 1, day);
          var dayEntries = byDate[key] || [];
          var isToday = year === currentYear && key === todayKey;
          var isFuture = year === currentYear && key > todayKey;
          var cell = el("div", "day" + (dayEntries.length ? " has-entry" : "") + (isToday ? " today" : "") + (isFuture ? " future" : ""));
          var number = el("div", "date-number", String(day));

          cell.appendChild(number);

          if (dayEntries.length) {
            applyDayStyle(cell, dayEntries, colorMap);
            cell.title = key + ": " + dayEntries.map(function (entry) {
              return entry.value;
            }).join(", ");

            if (dayEntries.length > 1) {
              cell.appendChild(el("span", "entry-count", String(dayEntries.length)));
            }
          }

          grid.appendChild(cell);
        }

        head.appendChild(title);
        head.appendChild(countLabel);
        month.appendChild(head);
        month.appendChild(weekdays);
        month.appendChild(grid);
        return month;
      }

      function applyDayStyle(cell, dayEntries, colorMap) {
        var dayValues = uniqueEntryValues(dayEntries);
        var colors = dayValues.map(function (value) {
          return colorMap[value];
        });

        cell.style.setProperty("--entry-color", colors[0].stroke);
        cell.style.setProperty("--entry-fill", colors[0].fill);

        if (colors.length > 1) {
          cell.style.background = activityGradient(colors);
          cell.style.borderColor = colors[0].stroke;
        }
      }

      function activityGradient(colors) {
        if (colors.length === 2) {
          return "linear-gradient(135deg, " + colors[0].fill + " 0 50%, " + colors[1].fill + " 50% 100%)";
        }

        var step = 100 / colors.length;
        return "conic-gradient(" + colors.map(function (color, index) {
          var start = (step * index).toFixed(2);
          var end = (step * (index + 1)).toFixed(2);
          return color.fill + " " + start + "% " + end + "%";
        }).join(", ") + ")";
      }

      function monthEntryCount(year, monthIndex, byDate) {
        var prefix = String(year) + "-" + pad2(monthIndex + 1) + "-";
        return Object.keys(byDate).reduce(function (sum, key) {
          return key.indexOf(prefix) === 0 ? sum + byDate[key].length : sum;
        }, 0);
      }

      function updateControls(year) {
        document.getElementById("year-title").textContent = String(year);
        document.getElementById("prev-year").disabled = false;
        document.getElementById("next-year").disabled = year >= currentYear;
      }

      function entriesByDateFor(selectedEntries) {
        var byDate = Object.create(null);
        selectedEntries.forEach(function (entry) {
          if (!byDate[entry.date]) {
            byDate[entry.date] = [];
          }
          byDate[entry.date].push(entry);
        });
        return byDate;
      }

      function valuesForEntries(selectedEntries) {
        var selectedValues = [];
        var seen = Object.create(null);
        selectedEntries.forEach(function (entry) {
          if (!seen[entry.value]) {
            seen[entry.value] = true;
            selectedValues.push(entry.value);
          }
        });
        return selectedValues;
      }

      function uniqueEntryValues(dayEntries) {
        var dayValues = [];
        var seen = Object.create(null);
        dayEntries.forEach(function (entry) {
          if (!seen[entry.value]) {
            seen[entry.value] = true;
            dayValues.push(entry.value);
          }
        });
        return dayValues;
      }

      function entryIsVisibleInYear(entry, year) {
        return entryYear(entry) === year && (year !== currentYear || entry.date <= todayKey);
      }

      function entryYear(entry) {
        return dateParts(entry.date).year;
      }

      function monthName(year, monthIndex) {
        return new Intl.DateTimeFormat(undefined, {
          month: "short"
        }).format(new Date(year, monthIndex, 1));
      }

      function dateParts(key) {
        var parts = key.split("-").map(Number);
        return {
          year: parts[0],
          month: parts[1],
          day: parts[2]
        };
      }

      function keyFromDate(date) {
        return dateKey(date.getFullYear(), date.getMonth() + 1, date.getDate());
      }

      function dateKey(year, month, day) {
        return String(year) + "-" + pad2(month) + "-" + pad2(day);
      }

      function pad2(number) {
        return String(number).padStart(2, "0");
      }

      function el(tag, className, text) {
        var node = document.createElement(tag);
        if (className) {
          node.className = className;
        }
        if (text !== undefined) {
          node.textContent = text;
        }
        return node;
      }
    }());
  </script>
